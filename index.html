CSS VARIABLE + FALLBACK STACK 
:root {
  --font-zad: "Zad",
    "Segoe UI", system-ui, -apple-system,
    "Helvetica Neue", Arial,
    "Noto Sans Arabic", sans-serif;
}

body { font-family: var(--font-zad); }
.font-zad { font-family: var(--font-zad); }
<!-- self-hosted: preload Regular -->
<link rel="preload" as="font"
      type="font/woff2" crossorigin
      href="zad-font/fonts/Zad-Regular.woff2">

<!-- jsDelivr: preconnect for faster TTFB -->
<link rel="preconnect"
      href="https://cdn.jsdelivr.net" crossorigin>
<!-- HTML root: pick a direction + lang -->
<html lang="ar" dir="rtl">
<html lang="en" dir="ltr">

/* CSS: numerals */
.tabular { font-variant-numeric: tabular-nums; }
.latin-figs   { font-feature-settings: "lnum" 1; }
/* Arabic-Indic digits: type the codepoints (٠–٩) */

<html lang="ar" dir="rtl">
